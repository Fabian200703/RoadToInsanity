country_event = {
	id = RSA_communist_civil_war.001
	title = RSA_soviet_communist_civil_war.001.t
	desc = RSA_soviet_communist_civil_war.001.desc
	picture = GFX_report_event_generic_battle
	is_triggered_only = yes
	hidden = no




	#SOV bekommen am 25.10.17 195, 208, 244

	immediate = {
		hidden_effect = { SOV_initiate_civil_war = yes }
	}
	trigger = { NOT = { has_global_flag = SOV_soviet_civil_war } }

	option = {
		name = RSA_communist_civil_war.001.a
		set_global_flag = flag_sided_with_lenin
		country_event = {
			id = RSA_communist_civil_war.002
			hours = 12
		}
	}
	option = {
		name = RSA_communist_civil_war.001.b
		set_global_flag = flag_sided_with_kerensky
		country_event = {
			id = RSA_communist_civil_war.003
			hours = 12
		}
	}
}
